import json, math

def wilson_ci(k, n, z=1.96):
    if n == 0: return 0.0, 100.0
    p = k / n
    denom = 1 + z**2/n
    center = (p + z**2/(2*n)) / denom
    margin = (z * math.sqrt(p*(1-p)/n + z**2/(4*n**2))) / denom
    return round(max(0, (center-margin)*100), 1), round(min(100, (center+margin)*100), 1)

def mean_ci(vals, z=1.96):
    n = len(vals)
    if n == 0: return 0.0, 0.0, 0.0
    m = sum(vals)/n
    if n == 1: return round(m,3), round(m,3), round(m,3)
    se = math.sqrt(max(sum((x-m)**2 for x in vals)/(n*(n-1)), 0))
    return round(m,3), round(m-z*se,3), round(m+z*se,3)
