df = pd.read_csv("https://marketwatch-jp.com/signals-log.csv")

# 銘柄 × 曜日 マトリクス
df.pivot_table(values="win", index="ticker", columns="発火_曜日", aggfunc="mean")

# 月末勝率（毎月最終 3 営業日）
df["fired_at"] = pd.to_datetime(df["fired_at"])
